peru23 = peru23 %>%
  mutate(educ = case_when(
    edre <= 3 ~ 1,
    edre == 4 ~ 2,
    edre == 5 ~ 3,
    edre == 6 ~ 4,
    ))
peru23 = peru23 %>%
  mutate(educ2 = factor(educ, labels=c("Colegio incompleto", 
                                       "Colegio completo", 
                                       "Superior incompleta",
